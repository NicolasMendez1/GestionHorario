CREATE TABLE NHMR_ROL(
    CODIGO NUMBER,
    NOMBRE_ROL VARCHAR2(50),
    PRIMARY KEY(CODIGO)
);

CREATE TABLE NHMR_USUARIO_GESTION(
    CODIGO NUMBER,
    RUT VARCHAR2(50),
    NOMBRE VARCHAR2(50),
    APELLIDO_PATERNO VARCHAR2(50),
    APELLIDO_MATERNO VARCHAR2(50),    
    CORREO VARCHAR2(50),
    CODIGO_ROL NUMBER,
    PRIMARY KEY(CODIGO),
    FOREIGN KEY(CODIGO_ROL) REFERENCES NHMR_ROL(CODIGO)
);
